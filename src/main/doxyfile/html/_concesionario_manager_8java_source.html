162 public ArrayList<Compra> getCompras() throws CompraException {
163 WebTarget donationsWebTarget = webTarget.path("collector/compra");
164 Invocation.Builder invocationBuilder = donationsWebTarget.request(MediaType.APPLICATION_JSON);
165 Response response = invocationBuilder.get();
166 if (response.getStatus() == Status.OK.getStatusCode()) {
167 ArrayList<Compra> compras = response.readEntity(ArrayList.class);
168 return compras;
169 } else {
170 throw new CompraException("" + response.getStatus());
171 }
